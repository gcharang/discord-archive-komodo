use cclib;
use cctokens;
use sixam; // hidden, standard included

module Foo {

    // vars
  char a;
  str b = "6am";  
    int c = 0;
    float d;
    bool e = false;
    address z;
    jdata jsonData;

    // key value pairs, arrays
    map(address => int) scores;
    map(address => str) names;
    list[str] scores;

    // constructor
    init Foo() {
        a = 'A';
        d = 22.5;
    }

    // public methods
    exec bar(str f) : public {

        // conditions with operators
        if(b.length == c) {
            barfoo();
        } else {
            foobar();
        }

    }

    exec boo() : public {
        // assign directly from response
        z = cctokens.tokenaddress().myaddress;
        jdata = cclib.games(5);
    }

    // private methods
    exec barfoo() : private {

        // iterating KV
        each(name in names) {
            // array manipulation
            scores.push(name);
        }

    }

    exec foobar() : private {
        // contract condition calls
        cclib.players(17);
    }

}```
