SHossain 29-Mar-19 11:21 AM
0. Install Dependencies
sudo apt-get update && sudo apt-get upgrade -y sudo apt-get install build-essential pkg-config libc6-dev m4 g++-multilib autoconf libtool ncurses-dev unzip git python zlib1g-dev wget bsdmainutils automake libboost-all-dev libssl-dev libprotobuf-dev protobuf-compiler libgtest-dev libqt4-dev libqrencode-dev libdb++-dev ntp ntpdate software-properties-common curl libcurl4-gnutls-dev cmake clang libsodium-dev -y
1. Clone & Compile
git clone https://github.com/jl777/komodo cd komodo git checkout jl777 ./zcutil/build.sh -j$(nproc) cd src/cc ./makegames ./maketetris
2. Start GTEST chain in a new terminal (use your pubkey)
cd ~/komodo/src ./komodod -ac_name=GTEST -ac_cclib=gamescc -ac_supply=1000000 -pubkey=<pubkey> -addnode=5.9.102.210 -ac_reward=100000000 -ac_cc=2 &
3. Copy GTEST.conf file using the first terminal or a new one
cd ~/komodo/src/cc/games cp ~/.komodo/GTEST/GTEST.conf ~/komodo/src/cc/games/GTEST.conf
4a. Start Tetris directly
cd ~/komodo/src/cc/games ./tetris
4b. Start Tetris in TUI
cd ~/komodo/src/tui ./tui_tetris.py
