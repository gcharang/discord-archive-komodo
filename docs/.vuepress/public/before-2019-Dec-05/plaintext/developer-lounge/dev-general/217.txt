Some elegant cpp snippet i write this evening for people interesting in `ranges` with modern c++

```cpp
 std::size_t system_manager::update_systems(system_type system_type_to_update) noexcept
    {
        std::size_t nb_systems_updated = 0ull;
        auto &&current_sys_collection = systems_[system_type_to_update];
        for (auto &&sys :  current_sys_collection | ranges::views::filter(&base_system::is_enabled)) {
            sys->update();
            nb_systems_updated += 1;
        }
        return nb_systems_updated;
    }
```
